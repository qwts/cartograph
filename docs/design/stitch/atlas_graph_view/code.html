pub async fn process_payment(
    req: PaymentRequest
) -> Result<PaymentResponse, Error> {
    
    // Inferred call to external provider
    let client = reqwest::Client::new();
    let res = client.post("https://api.stripe.com/v1/charges")
        .header("Authorization", format!("Bearer {}", API_KEY))
        .json(&req)
        .send()
        .await?;

    if res.status().is_success() {
        Ok(res.json().await?)
    } else {
        Err(Error::ProviderFailed)
    }
}
